47pub async fn run_tui(man_db: ManDb) -> Result<()> {
48    enable_raw_mode()?;
49    let mut stdout = std::io::stdout();
50    execute!(stdout, EnterAlternateScreen, EnableMouseCapture)?;
51    let backend = CrosstermBackend::new(stdout);
52    let mut terminal = Terminal::new(backend)?;
53
54    let man_db = Arc::new(man_db);
55    let commands = man_db.get_commands();
56    let filtered_commands = Arc::new(commands.clone());
57
58    let mut app = AppState {
59        input: String::new(),
60        filtered_commands,
61        selected_idx: 0,
62        man_content: Arc::new(Vec::new()),
63        scroll: 0,
64        focus: Focus::CommandList,
65        man_db: man_db.clone(),
66        loading: false,
67        search_query: String::new(),
68        search_matches: Arc::new(Vec::new()),
69        current_match: 0,
70        list_scroll: 0,
71        last_input_time: Instant::now(),
72        pending_man_load: true,
73        visible_list_range: (0, 0),
74    };
75
76    loop {
77        let now = Instant::now();
78
79        // Обработка отложенной загрузки man-страницы
80        if app.pending_man_load && app.last_input_time.elapsed() > Duration::from_millis(DEBOUNCE_DELAY_MS) {
81            load_current_man_page(&mut app).await;
82            app.pending_man_load = false;
83        }
84
85        terminal.draw(|f| render_ui(f, &mut app))?;
86
87        if event::poll(Duration::from_millis(16))? {
88            if let Event::Key(key) = event::read()? {
89                if key.kind != KeyEventKind::Press {
90                    continue;
91                }
92
93                match key.code {
94                    KeyCode::Char('q') => break,
95                    KeyCode::Tab => toggle_focus(&mut app),
96                    KeyCode::Esc => app.focus = Focus::CommandList,
97                    KeyCode::Char('/') if matches!(app.focus, Focus::ManPage) => {
98                        app.focus = Focus::Search;
99                        app.search_query.clear();
100                    }
101                    _ => handle_key(&mut app, key.code).await,
102                }
103            }
104        }
105
106        // Ограничение FPS для экономии ресурсов
107        let elapsed = now.elapsed();
108        if elapsed < Duration::from_millis(16) {
109            tokio::time::sleep(Duration::from_millis(16) - elapsed).await;
110        }
111    }
112
113    disable_raw_mode()?;
114    execute!(
115        terminal.backend_mut(),
116        LeaveAlternateScreen,
117        DisableMouseCapture
118    )?;
119    terminal.show_cursor()?;
120    Ok(())
121}

123fn toggle_focus(app: &mut AppState) {
124    app.focus = match app.focus {
125        Focus::CommandList => Focus::ManPage,
126        Focus::ManPage => Focus::CommandList,
127        Focus::Search => Focus::ManPage,
128    };
129}

131async fn handle_key(app: &mut AppState, key: KeyCode) {
132    match app.focus {
133        Focus::CommandList => handle_command_list_keys(app, key).await,
134        Focus::ManPage => handle_man_page_keys(app, key),
135        Focus::Search => handle_search_keys(app, key),
136    }
137}

139async fn handle_command_list_keys(app: &mut AppState, key: KeyCode) {
140    let commands_len = app.filtered_commands.len();
141    if commands_len == 0 {
142        return;
143    }
144
145    match key {
146        KeyCode::Char(c) => {
147            app.input.push(c);
148            filter_commands(app);
149            app.pending_man_load = true;
150            app.last_input_time = Instant::now();
151        }
152        KeyCode::Backspace => {
153            app.input.pop();
154            filter_commands(app);
155            app.pending_man_load = true;
156            app.last_input_time = Instant::now();
157        }
158        KeyCode::Up => {
159            if app.selected_idx > 0 {
160                app.selected_idx -= 1;
161                update_list_scroll(app);
162                app.pending_man_load = true;
163                app.last_input_time = Instant::now();
164            }
165        }
166        KeyCode::Down => {
167            if app.selected_idx < commands_len - 1 {
168                app.selected_idx += 1;
169                update_list_scroll(app);
170                app.pending_man_load = true;
171                app.last_input_time = Instant::now();
172            }
173        }
174        KeyCode::PageUp => {
175            app.selected_idx = app.selected_idx.saturating_sub(LIST_SIZE).max(0);
176            update_list_scroll(app);
177            app.pending_man_load = true;
178            app.last_input_time = Instant::now();
179        }
180        KeyCode::PageDown => {
181            app.selected_idx = (app.selected_idx + LIST_SIZE).min(commands_len - 1);
182            update_list_scroll(app);
183            app.pending_man_load = true;
184            app.last_input_time = Instant::now();
185        }
186        KeyCode::Enter => {
187            // Принудительная загрузка при нажатии Enter
188            app.pending_man_load = true;
189            load_current_man_page(app).await;
190            app.pending_man_load = false;
191        }
192        _ => {}
193    }
194}
195
196fn update_list_scroll(app: &mut AppState) {
197    let visible_height = app.visible_list_range.1 - app.visible_list_range.0;
198
199    if app.selected_idx < app.list_scroll {
200        app.list_scroll = app.selected_idx;
201    } else if app.selected_idx >= app.list_scroll + visible_height {
202        app.list_scroll = app.selected_idx - visible_height + 1;
203    }
204}

206fn filter_commands(app: &mut AppState) {
207    let commands = app.man_db.get_commands();
208
209    if app.input.is_empty() {
210        app.filtered_commands = Arc::new(commands.clone());
211    } else {
212        let filtered: Vec<String> = commands
213            .iter()
214            .filter(|cmd| cmd.starts_with(&app.input))
215            .cloned()
216            .collect();
217        app.filtered_commands = Arc::new(filtered);
218    }
219    app.selected_idx = 0;
220    app.list_scroll = 0;
221}

223async fn load_current_man_page(app: &mut AppState) {
224    if app.filtered_commands.is_empty() {
225        app.man_content = Arc::new(vec!["No commands found".to_string()]);
226        return;
227    }
228
229    let cmd = app.filtered_commands[app.selected_idx].clone();
230    app.loading = true;
231    let content = app.man_db.get_man_page(&cmd).await;
232    app.man_content = content;
233    app.loading = false;
234    app.scroll = 0;
235    update_search_matches(app);
236}

238fn handle_man_page_keys(app: &mut AppState, key: KeyCode) {
239    match key {
240        KeyCode::Up => app.scroll = app.scroll.saturating_sub(1),
241        KeyCode::Down => app.scroll = app.scroll.saturating_add(1),
242        KeyCode::PageUp => app.scroll = app.scroll.saturating_sub(PAGE_SIZE),
243        KeyCode::PageDown => app.scroll = (app.scroll + PAGE_SIZE).min(app.man_content.len().saturating_sub(PAGE_SIZE)),
244        KeyCode::Char('n') => next_search_match(app),
245        KeyCode::Char('N') => prev_search_match(app),
246        _ => {}
247    }
248}
249
250fn handle_search_keys(app: &mut AppState, key: KeyCode) {
251    match key {
252        KeyCode::Enter => {
253            update_search_matches(app);
254            app.focus = Focus::ManPage;
255        }
256        KeyCode::Char(c) => {
257            app.search_query.push(c);
258            update_search_matches(app);
259        }
260        KeyCode::Backspace => {
261            app.search_query.pop();
262            update_search_matches(app);
263        }
264        KeyCode::Esc => {
265            app.search_query.clear();
266            app.search_matches = Arc::new(Vec::new());
267            app.focus = Focus::ManPage;
268        }
269        _ => {}
270    }
271}
272
273fn update_search_matches(app: &mut AppState) {
274    let mut matches = Vec::new();
275
276    if !app.search_query.is_empty() {
277        for (i, line) in app.man_content.iter().enumerate() {
278            if line.contains(&app.search_query) {
279                matches.push(i);
280            }
281        }
282    }
283
284    app.search_matches = Arc::new(matches);
285    app.current_match = 0;
286
287    if !app.search_matches.is_empty() {
288        app.scroll = app.search_matches[0].saturating_sub(PAGE_SIZE / 2);
289    }
290}
291
292fn next_search_match(app: &mut AppState) {
293    if app.search_matches.is_empty() {
294        return;
295    }
296
297    app.current_match = (app.current_match + 1) % app.search_matches.len();
298    let target_line = app.search_matches[app.current_match];
299    app.scroll = target_line.saturating_sub(PAGE_SIZE / 2);
300}
301
302fn prev_search_match(app: &mut AppState) {
303    if app.search_matches.is_empty() {
304        return;
305    }
306
307    app.current_match = app.current_match.checked_sub(1)
308        .unwrap_or(app.search_matches.len() - 1);
309
310    let target_line = app.search_matches[app.current_match];
311    app.scroll = target_line.saturating_sub(PAGE_SIZE / 2);
312}

314fn render_ui<B: tui::backend::Backend>(f: &mut tui::Frame<B>, app: &mut AppState) {
315    let chunks = Layout::default()
316        .direction(Direction::Vertical)
317        .constraints(
318            [
319                Constraint::Length(1),
320                Constraint::Length(3),
321                Constraint::Min(10),
322            ]
323                .as_ref(),
324        )
325        .split(f.size());
326
327    render_status_bar(f, app, chunks[0]);
328    render_input(f, app, chunks[1]);
329    render_main_content(f, app, chunks[2]);
330}

332fn render_status_bar<B: tui::backend::Backend>(f: &mut tui::Frame<B>, app: &AppState, area: Rect) {
333    let status = if app.loading {
334        "Loading..."
335    } else {
336        match app.focus {
337            Focus::CommandList => "COMMAND LIST [Tab:Switch]",
338            Focus::ManPage => "MAN PAGE [Tab:Switch /:Search n/N:Next/Prev]",
339            Focus::Search => "SEARCH MODE [Enter:Apply Esc:Cancel]",
340        }
341    };
342
343    let status_bar = Paragraph::new(status)
344        .block(Block::default())
345        .style(Style::default().bg(Color::DarkGray));
346
347    f.render_widget(status_bar, area);
348}

350fn render_input<B: tui::backend::Backend>(f: &mut tui::Frame<B>, app: &AppState, area: Rect) {
351    let input_text = match app.focus {
352        Focus::CommandList | Focus::ManPage => format!("> {}", app.input),
353        Focus::Search => format!("/{}", app.search_query),
354    };
355
356    let input = Paragraph::new(input_text.as_str())
357        .block(Block::default().borders(Borders::ALL))
358        .style(Style::default().fg(Color::Yellow));
359
360    f.render_widget(input, area);
361}

363fn render_main_content<B: tui::backend::Backend>(f: &mut tui::Frame<B>, app: &mut AppState, area: Rect) {
364    let chunks = Layout::default()
365        .direction(Direction::Horizontal)
366        .constraints([Constraint::Percentage(30), Constraint::Percentage(70)].as_ref())
367        .split(area);
368
369    render_command_list(f, app, chunks[0]);
370    render_man_page(f, app, chunks[1]);
371}

373fn render_command_list<B: tui::backend::Backend>(f: &mut tui::Frame<B>, app: &mut AppState, area: Rect) {
374    let height = area.height as usize;
375    app.visible_list_range = (app.list_scroll, app.list_scroll + height);
376
377    let end = std::cmp::min(app.list_scroll + height, app.filtered_commands.len());
378    let visible_commands = &app.filtered_commands[app.list_scroll..end];
379
380    let items: Vec<ListItem> = visible_commands
381        .iter()
382        .map(|cmd| ListItem::new(cmd.as_str()))
383        .collect();
384
385    let list = List::new(items)
386        .block(Block::default().borders(Borders::ALL).title("Commands"))
387        .highlight_style(Style::default().bg(Color::DarkGray));
388
389    let mut state = ListState::default();
390    state.select(Some(app.selected_idx - app.list_scroll));
391    f.render_stateful_widget(list, area, &mut state);
392}

394fn render_man_page<B: tui::backend::Backend>(f: &mut tui::Frame<B>, app: &AppState, area: Rect) {
395    let height = area.height as usize;
396    let start_line = app.scroll;
397    let end_line = std::cmp::min(start_line + height, app.man_content.len());
398
399    let visible_content: Vec<Spans> = app.man_content
400        .iter()
401        .enumerate()
402        .skip(start_line)
403        .take(end_line - start_line)
404        .map(|(idx, line)| {
405            let global_idx = idx + start_line;
406            if app.search_matches.contains(&global_idx) {
407                let search_index = app.search_matches.iter().position(|&i| i == global_idx).unwrap();
408                let highlight = search_index == app.current_match;
409
410                let mut spans = Vec::new();
411                let mut remaining = line.as_str();
412
413                while let Some(pos) = remaining.find(&app.search_query) {
414                    let (before, after) = remaining.split_at(pos);
415                    let (match_text, rest) = after.split_at(app.search_query.len());
416
417                    spans.push(Span::raw(before));
418                    spans.push(Span::styled(
419                        match_text,
420                        Style::default()
421                            .bg(if highlight { Color::Red } else { Color::DarkGray })
422                            .fg(if highlight { Color::White } else { Color::Black })
423                    ));
424
425                    remaining = rest;
426                }
427                spans.push(Span::raw(remaining));
428
429                Spans::from(spans)
430            } else {
431                Spans::from(Span::raw(line.as_str()))
432            }
433        })
434        .collect();
435
436    let paragraph = Paragraph::new(visible_content)
437        .block(Block::default().borders(Borders::ALL).title("Man Page"))
438        .wrap(Wrap { trim: true });
439
440    f.render_widget(paragraph, area);
